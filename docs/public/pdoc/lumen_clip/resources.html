109    def load_model_resources(
110        cache_dir: str | Path, model_config: ModelConfig
111    ) -> ModelResources:
112        """
113        Load all resources for a model based on configuration intent.
114
115        Args:
116            cache_dir: Root cache directory from global config.
117            model_config: The specific model's configuration from lumen_config.
118
119        Returns:
120            ModelResources object with all loaded resources.
121        """
122        cache_dir = Path(cache_dir).expanduser().resolve()
123        model_root_path = cache_dir / "models" / model_config.model
124
125        logger.info(
126            f"Loading resources for {model_config.model} (runtime: {model_config.runtime.value})"
127        )
128
129        # Step 1: Load the on-disk manifest (model_info.json)
130        try:
131            model_info = lumen_resources.load_and_validate_model_info(
132                model_root_path / "model_info.json"
133            )
134        except FileNotFoundError:
135            raise ResourceNotFoundError(
136                f"model_info.json not found in {model_root_path}"
137            )
138        except ValidationError as e:
139            raise ModelInfoError(f"model_info.json is invalid: {e}") from e
140
141        # Step 2: Validate intent against the manifest and determine paths
142        runtime_files_path = ResourceLoader._validate_intent_and_get_paths(
143            model_root_path, model_config, model_info
144        )
145
146        # Step 3: Load config and tokenizer based on source format
147        source_format = model_info.source.format.value
148        if source_format == "openclip":
149            # Load model config.json (required) and attempt to load tokenizer.json (optional)
150            config = ResourceLoader._load_json(model_root_path / "open_clip_config.json")
151            tokenizer_config = ResourceLoader._load_for_openclip(model_root_path)
152            if tokenizer_config is None:
153                logger.info(
154                    "No tokenizer config loaded for OpenCLIP model; using SimpleTokenizer fallback"
155                )
156        elif source_format == "huggingface":
157            config = ResourceLoader._load_json(model_root_path / "config.json")
158            tokenizer_config = ResourceLoader._load_for_huggingface(model_root_path)
159        else:
160            raise ModelInfoError(
161                f"Unsupported source_format '{source_format}'. Must be 'openclip' or 'huggingface'."
162            )
163
164        # Step 4: Load dataset if specified
165        labels, embeddings = ResourceLoader._load_dataset(
166            model_root_path, model_info, model_config.dataset
167        )
168
169        source_repo = model_info.source.repo_id
170
171        resources = ModelResources(
172            model_root_path=model_root_path,
173            runtime_files_path=runtime_files_path,
174            model_name=model_config.model,
175            runtime=str(model_config.runtime.value),
176            model_info=model_info,
177            config=config,
178            tokenizer_config=tokenizer_config,
179            labels=labels,
180            label_embeddings=embeddings,
181            source_format=source_format,
182            source_repo=source_repo,
183        )
184
185        ResourceLoader._log_resource_summary(resources)
186        return resources

188    @staticmethod
189    def _validate_intent_and_get_paths(
190        model_root: Path, config: ModelConfig, info: ModelInfo
191    ) -> Path:
192        """
193        Validates the user's ModelConfig intent against the model_info manifest.
194        Returns the correct path to runtime files.
195        """
196        runtime = config.runtime.value
197        if runtime not in info.runtimes:
198            raise RuntimeNotSupportedError(
199                f"Runtime '{runtime}' not listed in model_info.json. "
200                f"Available: {list(info.runtimes.keys())}"
201            )
202        runtime_info = info.runtimes[runtime]
203        if not runtime_info.available:
204            raise RuntimeNotSupportedError(
205                f"Runtime '{runtime}' is not available for this model."
206            )
207
208        # Specific validation for RKNN runtime
209        if runtime == "rknn":
210            if not config.rknn_device:
211                raise ModelInfoError(
212                    "rknn_device must be specified in config for rknn runtime."
213                )
214            if runtime_info.devices and config.rknn_device not in runtime_info.devices:
215                raise RuntimeNotSupportedError(
216                    f"Device '{config.rknn_device}' not supported for this model. "
217                    f"Supported devices: {runtime_info.devices}"
218                )
219            runtime_path = model_root / runtime / config.rknn_device
220        elif runtime == "torch":
221            runtime_path = model_root
222        else:
223            runtime_path = model_root / runtime
224
225        if not runtime_path.exists():
226            raise ResourceNotFoundError(f"Runtime directory not found: {runtime_path}")
227
228        # Check for existence of required files for the runtime
229        if runtime_info.files and isinstance(runtime_info.files, list):
230            for file_path_str in runtime_info.files:
231                full_path = model_root / file_path_str
232                if not full_path.exists():
233                    raise ResourceNotFoundError(
234                        f"Required runtime file missing: {full_path}"
235                    )
236        return runtime_path

238    @staticmethod
239    def _load_json(path: Path) -> dict[str, Any]:
240        """Loads and parses a JSON file, raising specific errors."""
241        try:
242            with path.open("r", encoding="utf-8") as f:
243                return json.load(f)
244        except FileNotFoundError:
245            raise ResourceNotFoundError(f"Required file not found: {path}")
246        except json.JSONDecodeError as e:
247            raise ResourceValidationError(f"Invalid JSON in {path.name}: {e}") from e
248        except OSError as e:
249            raise ResourceNotFoundError(f"Failed to load {path.name}: {e}") from e

251    @staticmethod
252    def _load_for_openclip(model_path: Path) -> dict[str, Any] | None:
253        """Load tokenizer for OpenCLIP models. Fallback to SimpleTokenizer if not found."""
254        tokenizer_path = model_path / "tokenizer.json"
255        if tokenizer_path.exists():
256            try:
257                tokenizer_config = ResourceLoader._load_json(tokenizer_path)
258                logger.info("Loaded tokenizer.json for OpenCLIP model")
259                return tokenizer_config
260            except Exception as e:
261                logger.warning(
262                    f"Failed to load tokenizer.json: {e}. Fallback to SimpleTokenizer"
263                )
264        else:
265            logger.info(
266                "No tokenizer.json found, will use SimpleTokenizer for OpenCLIP model"
267            )
268        return None

270    @staticmethod
271    def _load_for_huggingface(model_path: Path) -> dict[str, Any] | None:
272        """Load tokenizer for HuggingFace models. tokenizer.json is required."""
273        tokenizer_path = model_path / "tokenizer.json"
274        if not tokenizer_path.exists():
275            logger.info("No tokenizer.json found, will use preprocessor.json for HuggingFace")
276            return None
277        try:
278            tokenizer_config = ResourceLoader._load_json(tokenizer_path)
279            logger.info("Loaded tokenizer.json for HuggingFace model")
280            return tokenizer_config
281        except Exception as e:
282            raise ResourceValidationError(
283                f"Failed to load tokenizer.json for HuggingFace model: {e}"

286    @staticmethod
287    def _load_dataset(
288        model_root_path: Path, model_info: ModelInfo, dataset: str | None
289    ) -> tuple[NDArray[np.object_] | None, NDArray[np.float32] | None]:
290
291        # 1. check if repo provides datasets
292        if not model_info.datasets:
293            logger.info(f"No datasets configured in model_info.json {model_info.datasets}")
294            return None, None
295
296        # 2. determine dataset name to load
297        dataset_name = dataset
298
299        # 3. validate dataset name
300        if not dataset_name or dataset_name not in model_info.datasets:
301            logger.warning(
302                f"Dataset '{dataset_name}' not in model_info.json. Disabling classification."
303            )
304            return None, None
305
306        # 4. get the dataset file relative paths
307        dataset_info: Datasets = model_info.datasets.get(dataset_name)
308        if not dataset_info:
309            logger.warning(
310                f"Dataset info for '{dataset_name}' is missing. Disabling classification."
311            )
312            return None, None
313        labels_path = model_root_path / dataset_info.labels
314        embeddings_path = model_root_path / dataset_info.embeddings
315
316        try:
317            labels: NDArray[np.object_] | None = None
318            embeddings: NDArray[np.float32] | None = None
319
320            if isinstance(dataset_info, Datasets):
321                if not labels_path.exists() or not embeddings_path.exists():
322                    raise FileNotFoundError(
323                        f"Missing dataset files: {labels_path} or {embeddings_path}"
324                    )
325
326                with open(labels_path, "r", encoding="utf-8") as f:
327                    labels = np.array(json.load(f), dtype=object)
328
329                logger.info(
330                    f"Loading embeddings for '{dataset_name}' with memory mapping..."
331                )
332                # FIX: Cast the result of np.load to the specific type we expect.
333                raw_embeddings = np.load(embeddings_path, mmap_mode="r")
334                embeddings = cast(NDArray[np.float32], raw_embeddings)
335
336            elif isinstance(dataset_info, str):
337                logger.warning(
338                    "Using legacy single-file dataset format. Memory mapping is not guaranteed."
339                )
340                dataset_path = model_root_path / dataset_info
341                if not dataset_path.exists():
342                    raise FileNotFoundError(f"Dataset file not found: {dataset_path}")
343
344                data = np.load(dataset_path, allow_pickle=True)
345                if isinstance(data, np.lib.npyio.NpzFile):
346                    # FIX: Cast the arrays from the NpzFile.
347                    labels = cast(NDArray[np.object_], data["labels"])
348                    if "embeddings" in data.files:
349                        embeddings = cast(NDArray[np.float32], data["embeddings"])
350                    data.close()
351                else:
352                    data_dict = data.item()
353                    labels = np.array(data_dict["labels"], dtype=object)
354                    if "embeddings" in data_dict:
355                        embeddings = cast(NDArray[np.float32], data_dict["embeddings"])
356            else:
357                raise TypeError(
358                    f"Unsupported format for dataset info: {type(dataset_info)}"
359                )
360
361            if labels is not None:
362                logger.info(f"Loaded dataset '{dataset_name}': {len(labels)} classes")
363
364            return labels, embeddings
365
366        except Exception as e:
367            logger.warning(
368                f"Failed to load dataset '{dataset_name}': {e}. Disabling classification."
369            )
370            return None, None
