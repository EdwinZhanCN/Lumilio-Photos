174        """

184        for req in request_iterator:

196    def _process_batch(
197        self, batch: list[pb.InferRequest]
198    ) -> Iterable[pb.InferResponse]:
199        """Processes a batch of requests, groups them by task, and yields responses."""
200        t0 = time.time()
201        logger.info(f"Processing a batch of {len(batch)} requests...")
202
203        # Group requests by their task name
204        tasks: dict[str, list[pb.InferRequest]] = defaultdict(list)
205        for req in batch:
206            tasks[req.task].append(req)
207
208        # Process each task group
209        for task_name, requests in tasks.items():
210            handler = getattr(self, f"_handle_{task_name}", None)
211            if handler:
212                try:
213                    for response in handler(requests):
214                        yield response
215                except Exception as e:
216                    logger.exception(f"Error processing task '{task_name}': {e}")
217                    # Yield an error response for all failed requests in this group
218                    for req in requests:
219                        yield pb.InferResponse(
220                            correlation_id=req.correlation_id,
221                            is_final=True,
222                            error=pb.Error(code=pb.ERROR_CODE_INTERNAL, message=str(e)),
223                        )
224            else:
225                logger.warning(f"Unknown task received: {task_name}")
226                for req in requests:
227                    yield pb.InferResponse(
228                        correlation_id=req.correlation_id,
229                        is_final=True,
230                        error=pb.Error(
231                            code=pb.ERROR_CODE_INVALID_ARGUMENT,
232                            message=f"Unknown task: {task_name}",
233                        ),
234                    )
235
236        processing_time = (time.time() - t0) * 1000
237        logger.info(f"Batch processing finished in {processing_time:.2f} ms.")

261    def _handle_smart_classify(
262        self, requests: list[pb.InferRequest]
263    ) -> Iterable[pb.InferResponse]:
264        for req in requests:
265            # Check if both models support classification
266            if not self.clip_model.supports_classification:
267                yield pb.InferResponse(
268                    correlation_id=req.correlation_id,
269                    is_final=True,
270                    error=pb.Error(
271                        code=pb.ERROR_CODE_INVALID_ARGUMENT,
272                        message="smart_classify requires CLIP classification support (no ImageNet dataset)",
273                    ),
274                )
275                continue
276
277            # 1) Scene classification using CLIP
278            scene_label, scene_score = self.clip_model.classify_scene(req.payload)
279
280            # Check if it's animal-like based on scene classification
281            is_animal_like = (
282                ("animal" in scene_label)
283                or ("bird" in scene_label)
284                or ("insect" in scene_label)
285            )
286
287            if not is_animal_like:
288                yield self._build_label_response(
289                    req.correlation_id,
290                    [(scene_label, scene_score)],
291                    self.clip_model.info(),
292                    meta={"source": "scene_classification"},
293                )
294                continue
295
296            # 2) Animal-like: classify with BioCLIP (if supported)
297            if not self.bioclip_model.supports_classification:
298                # Fallback to CLIP classification
299                yield self._build_label_response(
300                    req.correlation_id,
301                    [(scene_label, scene_score)],
302                    self.clip_model.info(),
303                    meta={"source": "scene_classification_fallback"},
304                )
305                continue
306
307            top_k = int(req.meta.get("topk", "3"))
308            scores = self.bioclip_model.classify_image(req.payload, top_k)
309            yield self._build_label_response(
310                req.correlation_id,
311                scores,
312                self.bioclip_model.info(),
313                meta={"source": "bioclip_classification"},
314            )

362        cid: str,
363        scores: list[tuple[str, float]],
364        model_info: Mapping[str, Any],
365        meta: dict[str, str] | None = None,
366    ) -> pb.InferResponse:
367        model_id = f"{model_info.get('model_name', model_info.get('model_version'))}:{model_info.get('pretrained', '')}"
368        obj = {
369            "labels": [{"label": label, "score": score} for label, score in scores],
370            "model_id": model_id.strip(":"),
371        }

373        if meta:
374            response_meta.update(meta)
375        return pb.InferResponse(

377            is_final=True,
378            result=json.dumps(obj, separators=(",", ":")).encode("utf-8"),
379            result_mime="application/json;schema=labels_v1",
