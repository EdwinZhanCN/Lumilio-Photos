127    def _compute_text_embeddings(self) -> None:
128        """
129        Compute text embeddings for all labels (called if not pre-computed).
130        """
131        if not self.labels:
132            logger.warning("No labels available, skipping embedding computation")
133            return
134
135        logger.info(f"Computing text embeddings for {len(self.labels)} species...")
136
137        # Use batch processing if backend supports it
138        prompts = [f"a photo of {name}" for name in self.labels]
139
140        try:
141            # Try batch processing first
142            embeddings_array = self.backend.text_batch_to_vectors(prompts)
143            self.text_embeddings = embeddings_array.astype(np.float32, copy=False)
144            logger.info("Text embeddings computed successfully")
145        except Exception as e:
146            logger.warning(f"Batch processing failed: {e}, falling back to sequential")
147            # Fallback to sequential processing
148            embeddings_list: list[NDArray[np.float32]] = []
149            for prompt in prompts:
150                embedding = self.backend.text_to_vector(prompt)
151                embeddings_list.append(embedding)
152            self.text_embeddings = np.vstack(embeddings_list).astype(
153                np.float32, copy=False
154            )

190    def extract_scientific_name(label_data: str | list[Any]) -> str:
191        """
192        Extract the scientific name from the complex TreeOfLife label structure.
193
194        Args:
195            label_data: The label data from TreeOfLife-10M
196

198            The scientific name as a string

202            and len(label_data) == 2
203            and isinstance(label_data[0], list)
204        ):
205            # Format: [['Animalia', ..., 'Genus', 'species'], 'Common Name']
206            taxonomy = label_data[0]

213    def classify_image(
214        self, image_bytes: bytes, top_k: int = 3
215    ) -> list[tuple[str, float]]:
216        """
217        Classify an image against TreeOfLife species labels.
218
219        Args:
220            image_bytes: Raw image data in bytes
221            top_k: Number of top results to return
222
223        Returns:
224            List of (scientific_name, probability) tuples sorted by probability (descending)
225
226        Raises:
227            RuntimeError: If model is not initialized or classification not supported
228        """
229        self._ensure_initialized()
230
231        if not self.supports_classification:
232            raise RuntimeError(
233                "Classification not supported: no dataset loaded. "
234                "Ensure TreeOfLife dataset exists in the model directory."
235            )
236
237        if self.text_embeddings is None:
238            raise RuntimeError("Text embeddings are not available")
239
240        # Get image embedding via backend
241        img_embedding = self.encode_image(image_bytes)
242
243        # Compute similarities using numpy (standard operations)
244        img_embedding = img_embedding / np.linalg.norm(
245            img_embedding
246        )  # Ensure unit norm
247        text_embeddings_norm = self.text_embeddings / np.linalg.norm(
248            self.text_embeddings, axis=1, keepdims=True
249        )
250
251        # Compute cosine similarities
252        similarities = np.dot(img_embedding, text_embeddings_norm.T)
253
254        # Convert to probabilities (softmax)
255        exp_sims = np.exp(similarities - np.max(similarities))  # Numerical stability
256        probabilities = exp_sims / np.sum(exp_sims)
257
258        # Get top-k results
259        top_indices = np.argsort(probabilities)[::-1][:top_k]
260
261        # Extract scientific names from the label data
262        results = [
263            (self.extract_scientific_name(self.labels[idx]), float(probabilities[idx]))
264            for idx in top_indices
265        ]
266
267        return results

 98        # Get dataset name if specified

132        service = cls(backend, resources)

134        # Log classification support status
135        if not resources.has_classification_support():
136            logger.warning(
137                "BioCLIP service started without classification support "

139            )

156    @override
157    def Infer(
158        self, request_iterator: Iterable[pb.InferRequest], context: grpc.ServicerContext
159    ):
160        """
161        Bidirectional streaming server implementation (synchronous style: generator yields responses on demand).
162        Both embed and classify are one-request-one-response; supports reassembly if client uses seq/total for fragmentation.
163        """
164        if not self.is_initialized:
165            context.abort(grpc.StatusCode.FAILED_PRECONDITION, "Model not initialized")
166
167        buffers: dict[str, bytearray] = {}  # correlation_id -> buffer
168
169        for req in request_iterator:
170            cid = req.correlation_id or f"cid-{_now_ms()}"
171            t0 = _now_ms()
172
173            try:
174                # --- Fragment reassembly (optional) ---
175                payload, ready = self._assemble(cid, req, buffers)
176                if not ready:
177                    # Fragment not yet complete; wait for additional fragments
178                    continue
179
180                # --- Route to task handler ---
181                if req.task == "embed":
182                    result_bytes, result_mime, extra_meta = self._handle_embed(
183                        req.payload_mime, payload, dict(req.meta)
184                    )
185                elif req.task == "classify":
186                    # Check if classification is supported
187                    if not self.model.supports_classification:
188                        yield pb.InferResponse(
189                            correlation_id=cid,
190                            is_final=True,

264        """
265        Text embedding:
266          - Expects payload_mime: "text/plain;charset=utf-8"
267          - Output result_mime: "application/json;schema=embedding_v1"

271            raise ValueError(f"embed expects text/* payload, got {payload_mime!r}")

276        obj = {"vector": vec, "dim": len(vec), "model_id": model_id}
277        return (
278            json.dumps(obj, separators=(",", ":")).encode("utf-8"),
279            "application/json;schema=embedding_v1",
280            {"dim": str(len(vec))},
281        )

307    def _handle_classify(
308        self, payload_mime: str, payload: bytes, meta: dict[str, str]
309    ) -> tuple[bytes, str, dict[str, str]]:
310        """
311        TreeOfLife classification:
312          - Expects payload_mime: "image/jpeg" / "image/png"
313          - Expects meta.namespace="bioatlas"
314          - Optional meta.topk (default 5)
315          - Output result_mime: "application/json;schema=labels_v1"
316          - Output JSON: {"labels":[{"label":"...","score":0.91},...], "model_id":"bioclip2"}
317        """
318        if not payload_mime.startswith("image/"):
319            raise ValueError(f"classify expects image/* payload, got {payload_mime!r}")
320
321        namespace = (meta or {}).get("namespace", "bioatlas")
322        if namespace != "bioatlas":
323            raise ValueError(
324                f"unsupported namespace {namespace!r}, expected 'bioatlas'"
325            )
326
327        topk = int((meta or {}).get("topk", "5"))
328        pairs = self.model.classify_image(
329            payload, top_k=topk
330        )  # list[tuple[str, float]]
331
332        info = self.model.info()
333        model_id = info.get("model_id", self.MODEL_VERSION)
334        obj = {
335            "labels": [{"label": name, "score": float(score)} for name, score in pairs],
336            "model_id": model_id,
337        }
338        return (
339            json.dumps(obj, separators=(",", ":")).encode("utf-8"),
340            "application/json;schema=labels_v1",
341            {"labels_count": str(len(pairs))},
342        )

344    # -------- buffers / 分片重组 ----------
345    def _assemble(
346        self, cid: str, req: pb.InferRequest, buffers: dict[str, bytearray]
347    ) -> tuple[bytes, bool]:
348        """
349        Return a tuple (payload_bytes, ready).
350
351        - If the client does not use seq/total: ready is True immediately.
352        - If seq/total are used: buffer fragments in buffers[cid] until all
353          fragments are received (seq + 1 == total); then ready becomes True.
354        """
355        # No fragmentation (default path)
356        if not req.total and not req.seq and not req.offset:
357            return bytes(req.payload), True
358
359        buf = buffers.setdefault(cid, bytearray())
360        buf.extend(req.payload)
361        if req.total and (req.seq + 1 == req.total):
362            data = bytes(buf)
363            del buffers[cid]
364            return data, True
365        return b"", False
